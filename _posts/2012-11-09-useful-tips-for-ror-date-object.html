Date Object Tips

  # 当天
  today = Date.today

  # 明天
  today.next

  # 本周一
  today.beginning_of_week

  # 下周一
  today.next_week

  # 7 天前, 一周前
  7.days.ago.to_date

  # 一周中的第几天：周日至周六，对应 0 至 6
  today.wday

  # 本月一号
  Date.today.beginning_of_month

  # 本月底
  Date.today.end_of_month

  # 下个月
  1.month.since

  # 两个小时后
  2.hours.from_now
